> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Expose the core mh_tools library via a FastAPI REST API, and build a Next.js frontend for browser-based access to the Map Value Analyser.

**Architecture:** FastAPI wraps the same `MapValueAnalyser`, `Database`, and provider classes from Phase 1-3. The Next.js frontend calls the API endpoints. Both can be run locally or deployed independently.

**Tech Stack:** FastAPI, uvicorn, Next.js (App Router), TypeScript, Tailwind CSS

**Prerequisite:** Phase 1-3 must be complete and all tests passing. Phase 4 (TUI) is independent and not required.

from __future__ import annotations

from fastapi import APIRouter, Depends, Query

from mh_tools.providers.mhct import MHCTProvider
from mh_tools.web.dependencies import get_mhct

router = APIRouter()


@router.get("/search")
def search_chests(
    q: str = Query(default="", description="Search query for chest names"),
    mhct: MHCTProvider = Depends(get_mhct),
) -> list[dict]:
    """Search for chests by name substring."""
    if not q:
        return mhct.list_convertibles()
    return mhct.search_convertible(q)

"""Analysis endpoints."""

from __future__ import annotations

from fastapi import APIRouter, Depends, HTTPException, Query

from mh_tools.database import Database
from mh_tools.providers.markethunt import MarketHuntProvider
from mh_tools.providers.mhct import MHCTProvider
from mh_tools.tools.map_value import MapValueAnalyser
from mh_tools.web.dependencies import get_db, get_markethunt, get_mhct

router = APIRouter()


@router.get("/map")
def analyse_map(
    chest_name: str = Query(description="Name of the chest to analyse"),
    db: Database = Depends(get_db),
    mhct: MHCTProvider = Depends(get_mhct),
    markethunt: MarketHuntProvider = Depends(get_markethunt),
) -> dict:
    """Analyse the expected value of a treasure chest."""
    analyser = MapValueAnalyser(db=db, mhct=mhct, markethunt=markethunt)
    try:
        result = analyser.run(chest_name=chest_name)
    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))
    return result.model_dump()
```

const API_BASE = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

export interface ChestItem {
  id: number;
  name: string;
}

export interface ItemEV {
  item_name: string;
  drop_chance: number;
  avg_quantity: number;
  gold_price: number | null;
  sb_price: number | null;
  ev_gold: number;
  ev_sb: number;
  unmapped: boolean;
  price_stale: boolean;
}

export interface AnalysisResult {
  chest_name: string;
  items: ItemEV[];
  total_ev_gold: number;
  total_ev_sb: number;
  total_ev_gold_after_tax: number;
  total_ev_sb_after_tax: number;
  sb_rate: number;
  warnings: string[];
}

export async function searchChests(
  query: string,
  signal?: AbortSignal,
): Promise<ChestItem[]> {
  const resp = await fetch(
    `${API_BASE}/chests/search?q=${encodeURIComponent(query)}`,
    { signal },
  );
  if (!resp.ok) throw new Error("Failed to search chests");
  return resp.json();
}

export async function analyseChest(chestName: string): Promise<AnalysisResult> {
  const resp = await fetch(
    `${API_BASE}/analysis/map?chest_name=${encodeURIComponent(chestName)}`,
  );
  if (!resp.ok) throw new Error("Failed to analyse chest");
  return resp.json();
}
```

`frontend/src/app/components/ChestSearch.tsx`:
```tsx
"use client";

import { useState, useEffect, useRef } from "react";
import { ChestItem, searchChests } from "@/lib/api";

interface Props {
  onSelect: (chest: ChestItem) => void;
}

export default function ChestSearch({ onSelect }: Props) {
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<ChestItem[]>([]);
  const [loading, setLoading] = useState(false);
  const abortRef = useRef<AbortController | null>(null);

  useEffect(() => {
    const timer = setTimeout(async () => {
      if (query.length < 2) {
        setResults([]);
        return;
      }
      // Cancel any in-flight request
      abortRef.current?.abort();
      const controller = new AbortController();
      abortRef.current = controller;

      setLoading(true);
      try {
        const data = await searchChests(query, controller.signal);
        setResults(data);
      } catch (err) {
        if (err instanceof DOMException && err.name === "AbortError") return;
        setResults([]);
      } finally {
        setLoading(false);
      }
    }, 300); // debounce
    return () => {
      clearTimeout(timer);
      abortRef.current?.abort();
    };
  }, [query]);

  return (
    <div className="w-full max-w-md">
      <input
        type="text"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="Search chests..."
        className="w-full p-2 border rounded bg-gray-800 text-white border-gray-600"
      />
      {loading && <p className="text-sm text-gray-400 mt-1">Searching...</p>}
      {results.length > 0 && (
        <ul className="mt-1 border rounded border-gray-600 max-h-60 overflow-y-auto">
          {results.map((chest) => (
            <li
              key={chest.id}
              onClick={() => onSelect(chest)}
              className="p-2 hover:bg-gray-700 cursor-pointer text-sm"
            >
              {chest.name}
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```

import { AnalysisResult } from "@/lib/api";

interface Props {

export default function AnalysisTable({ result, showSB, showAfterTax }: Props) {
  const totalEV = showAfterTax
    ? showSB ? result.total_ev_sb_after_tax : result.total_ev_gold_after_tax
    : showSB ? result.total_ev_sb : result.total_ev_gold;

  const unit = showSB ? "SB" : "Gold";

  return (
    <div className="w-full">
      <h2 className="text-xl font-bold mb-2">{result.chest_name}</h2>
      <p className="text-sm text-gray-400 mb-4">
        SB Rate: {result.sb_rate.toLocaleString()} gold/SB

        <thead>

            <th className="text-right p-2">Avg Qty</th>
            <th className="text-right p-2">Price</th>
            <th className="text-right p-2">EV ({unit})</th>
          </tr>
        </thead>
        <tbody>
          {result.items.map((item) => (
